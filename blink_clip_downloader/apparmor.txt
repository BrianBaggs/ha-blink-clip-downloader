#include <tunables/global>

profile blink_clip_downloader flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/python>

  # Network access for Blink API and HA supervisor
  network tcp,
  network udp,

  # Python runtime
  /usr/bin/python3* mrix,
  /usr/local/lib/python3*/** mr,
  /usr/lib/python3*/** mr,

  # App package installed via pip
  /usr/local/lib/python3*/dist-packages/blink_downloader/** r,
  /usr/local/lib/python3*/dist-packages/blinkpy/** r,

  # Persistent data directory
  /data/ r,
  /data/** rw,

  # Downloaded clips and share mount
  /share/ r,
  /share/** rw,

  # Proc and sys (read-only minimal access)
  @{PROC}/@{pid}/fd/ r,
  @{PROC}/@{pid}/status r,
  @{PROC}/sys/kernel/random/uuid r,

  # Deny everything else
  deny /etc/** w,
  deny /bin/** wl,
  deny /sbin/** wl,
  deny /root/** rw,
}
